# Soccer AI Analytics - Motor de Detecção import cv2 from ultralytics import YOLO # 1. Carregar o modelo de IA (YOLOv8 é o mais rápido para futebol) model = YOLO('yolov8n.pt') def process_video(video_path): # 2. Abrir o vídeo da partida cap = cv2.VideoCapture(video_path) while cap.isOpened(): ret, frame = cap.read() if not ret: break # 3. IA detectando jogadores (class 0) e bola (class 32) results = model(frame, classes=[0, 32]) # Aqui a IA desenha as caixas nos jogadores em tempo real annotated_frame = results[0].plot() # Exibir o resultado (apenas se rodar no PC) cv2.imshow("IA Soccer Analysis", annotated_frame) if cv2.waitKey(1) & 0xFF == ord('q'): break cap.release() cv2.destroyAllWindows() print("Motor de IA configurado com sucesso!")
